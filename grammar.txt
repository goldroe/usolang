
IDENT    = [a-zA-Z][_0-9a-zA-Z]*
CHARLIT  = \' ([^\n\t\\\'] | \\' | \\\\ | \\n | \\t) \'
STRLIT   = \" ([^\n\t\\\"] | \\" | \\\\ | \\n | \\t)* \"

FLOATLIT = (0|[1-9][0-9]*)f
         | .[0-9]*f
         | (0|[1-9][0-9]*).[0-9]*f

INTLIT   = 0 | [1-9][0-9]*    DECIMAL
         | 0[xX][0-9a-fA-F]+  HEX
         | 0[0-7]+            OCTAL
         | 0[bB][0-1]+        BINARY

COMMENTS = // * (\r\n|\r|\n|)
         | /\* ([^\*]|(\*+[^/])*? \*/
